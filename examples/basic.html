var N = 100, E = 500, nodes = [], edges = [];

// Generate a random graph:
for (var i = 0; i < N; i++)
  nodes.push({
    id: 'n' + i,
    label: 'Node ' + i,
    x: Math.random(),
    y: Math.random(),
    size: Math.random(),
    color: '#666'
  });

for (var i = 0; i < E; i++)
  edges.push({
    id: 'e' + i,
    source: 'n' + (Math.random() * N | 0),
    target: 'n' + (Math.random() * N | 0),
    size: Math.random(),
    color: '#ccc'
  });

// Instantiate sigma:
var s = new sigma({
  graph: {nodes:nodes, edges:edges},
  container: 'graph-container'
});
